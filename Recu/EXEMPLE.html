Să scriem o funcție C++ care returnează factorialul unui număr natural transmis ca parametru. Varianta nerecursivă (iterativă) este următoarea:
int fact(int n){
int p = 1;
for(int i = 1 ; i <= n ; i ++)
p = p * i;
return p;
}

FUNCTIE DE TIP VOID:

int fact(int n){
int f;
if(n == 0)
return 1;
else
f = fact(n - 1) * n;
return f;
}
int main(){
int x = fact(3);
cout << x;
return 0;
}
FUNCTIE DE TIP NON-VOID:
int fact(int n){
int r;
if(n == 0)
r = 1;
else
r = n * fact(n - 1);
return r;
}
int main(){
int a;
a = fact(4);
cout << a;
return 0;
}
